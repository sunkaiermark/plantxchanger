# PlantXchange Next.js + Strapi Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build PlantXchange v1 as a Next.js public catalog backed by Strapi CMS content and Strapi-stored inquiry records.

**Architecture:** Use an npm-workspaces monorepo with `apps/web` for the Next.js App Router site and `apps/cms` for the Strapi 5 CMS. Next.js reads published catalog content from Strapi on the server, submits inquiries through a server-side route handler, and never exposes Strapi write tokens to the browser.

**Tech Stack:** npm workspaces, Next.js App Router, React, TypeScript, Tailwind CSS, Strapi 5, SQLite for local Strapi development, Strapi REST API, Node test runner with `tsx`.

# PlantXchange CMS

Strapi stores PlantXchange equipment content, categories, media, site settings, and inquiry records.

## Local Setup

```powershell
Copy-Item .env.example .env

Create two tokens in Strapi Admin:

- Write token: create access for Inquiry records only.

Store the token values in `apps/web/.env.local`; never expose them in browser code.

const FALLBACK_IMAGE = "/images/chemical-plant.png";

export function normalizeMediaUrl(media: MediaLike | null | undefined, strapiUrl: string): string {
  const url = media?.url;
  if (!url) return FALLBACK_IMAGE;
  if (url.startsWith("http://") || url.startsWith("https://")) return url;
  return `${strapiUrl.replace(/\/$/, "")}${url}`;
}

export function normalizeEquipment(raw: any, strapiUrl: string): EquipmentSummary {
  return {
    documentId: String(raw.documentId ?? raw.id ?? ""),
    title: String(raw.title ?? ""),
    slug: String(raw.slug ?? ""),
    reference: String(raw.reference ?? ""),
    condition: String(raw.condition ?? "good"),
    availability: String(raw.availability ?? "available"),
    country: optionalString(raw.country),
    location: optionalString(raw.location),
    year: optionalNumber(raw.year),
    make: optionalString(raw.make),
    model: optionalString(raw.model),
    price: optionalNumber(raw.price),
    currency: optionalString(raw.currency),
    summary: optionalString(raw.summary),
    description: optionalString(raw.description),
    specifications: Array.isArray(raw.specifications) ? raw.specifications : [],
    features: Array.isArray(raw.features) ? raw.features : [],
    category: {
      name: String(raw.category?.name ?? "Uncategorized"),
      slug: String(raw.category?.slug ?? "uncategorized"),
      description: optionalString(raw.category?.description),
    },
    mainImageUrl: normalizeMediaUrl(raw.mainImage, strapiUrl),
    galleryUrls: Array.isArray(raw.gallery)
      ? raw.gallery.map((item: MediaLike) => normalizeMediaUrl(item, strapiUrl))
      : [],

    isFeatured: Boolean(raw.isFeatured),
    seoTitle: optionalString(raw.seoTitle),
    seoDescription: optionalString(raw.seoDescription),
  };

function optionalString(value: unknown): string | undefined {

```ts
import qs from "qs";
import { getRequiredServerEnv } from "@/lib/env";

type StrapiRequestMode = "read" | "write";

export function getStrapiUrl(): string {
  return getRequiredServerEnv("STRAPI_URL").replace(/\/$/, "");
}

export async function strapiFetch<T>(
  path: string,
  options: {
    mode?: StrapiRequestMode;
    query?: Record<string, unknown>;
    init?: RequestInit;
    revalidate?: number;
  } = {},
): Promise<T> {
  const mode = options.mode ?? "read";
  const token = getRequiredServerEnv(mode === "read" ? "STRAPI_READ_TOKEN" : "STRAPI_WRITE_TOKEN");
  const query = options.query ? `?${qs.stringify(options.query, { encodeValuesOnly: true })}` : "";
  const url = `${getStrapiUrl()}${path}${query}`;

  const response = await fetch(url, {
    ...options.init,
    headers: {
      Authorization: `Bearer ${token}`,
      "Content-Type": "application/json",
      ...(options.init?.headers ?? {}),
    },
    next: options.revalidate ? { revalidate: options.revalidate } : undefined,
  });

  if (!response.ok) {
    const text = await response.text();
    throw new Error(`Strapi ${response.status} for ${path}: ${text.slice(0, 240)}`);
  }

  return (await response.json()) as T;
}

      pagination: { pageSize: 48 },

const inquirySchema = z

    utmMedium: z.string().trim().max(120).optional(),
    utmCampaign: z.string().trim().max(120).optional(),
    website: z.string().optional(),

  .superRefine((value, context) => {
    if (value.website && value.website.trim() !== "") {

import { NextResponse } from "next/server";
import { strapiFetch } from "@/lib/strapi/client";
import { buildInquiryCreatePayload } from "@/lib/inquiries/payload";
import { validateInquiryInput } from "@/lib/inquiries/validation";

export async function POST(request: Request) {
  const json = await request.json().catch(() => null);
  const parsed = validateInquiryInput(json);

  if (!parsed.success) {
    return NextResponse.json(
      { ok: false, message: "Please check the inquiry form and try again." },
      { status: 400 },
    );
  }

  try {
    await strapiFetch("/api/inquiries", {
      mode: "write",
      init: {
        method: "POST",
        body: JSON.stringify(
          buildInquiryCreatePayload(parsed.data, {
            userAgent: request.headers.get("user-agent"),
            ipAddress:
              request.headers.get("x-forwarded-for")?.split(",")[0]?.trim() ??
              request.headers.get("x-real-ip"),
          }),
        ),
      },
    });

    return NextResponse.json({ ok: true });
  } catch {
    return NextResponse.json(
      {
        ok: false,
        message: "We could not save the inquiry. Please contact us by email or WhatsApp.",
      },
      { status: 502 },
    );
  }
}
